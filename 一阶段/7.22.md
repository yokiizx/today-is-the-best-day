20. 利用栈解决即可,注意 map 设置的顺序

```js
var isValid = function (s) {
  // 栈解决，如果最后栈没有清空就是不合格的
  const stack = []
  const map = {
    ')': '(',
    '}': '{',
    ']': '['
  }
  for (const c of String(s)) {
    if (Object.values(map).includes(c)) {
      stack.push(c)
    } else {
      if (stack[stack.length - 1] === map[c]) {
        stack.pop()
      } else {
        stack.push(c)
      }
    }
  }
  return stack.length === 0
}
```

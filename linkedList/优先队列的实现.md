二叉堆在逻辑上其实是一种特殊的二叉树（完全二叉树），只不过存储在数组里。一般的链表二叉树，我们操作节点的指针，而在数组里，我们把数组索引作为指针

**数组的第一个索引 0 空着不用**

优先级队列这种数据结构有一个很有用的功能，你插入或者删除元素的时候，元素会自动排序，这底层的原理就是二叉堆的操作。

```js
class PriorityQueue {
  constructor() {
    this.heap = []
  }
  // 获取节点关系
  getParent(i) {
    // return ((i + 1) >>> 1) - 1
    return ((i - 1) / 2) | 0
  }
  getLeft(i) {
    return 2 * i + 1
  }
  getRight(i) {
    return 2 * i + 2
  }
  // 交换元素位置，优先队列中常用于交换堆顶和堆尾
  swap(i, j) {
    const temp = this.heap[i]
    this.heap[i] = this.heap[j]
    this.heap[j] = temp
  }
  // 比较 此处实现是smallerThanParent，改变小于号为大于号，即可改变堆的大小
  compare(i, j) {
    return this.heap[i] - this.heap[j] < 0
  }
  // 上浮 第i个元素，与父节点比较大小即可
  shiftUp(i) {
    // 上浮到顶就停止
    while (i > 0 && this.compare(i, this.getParent(i))) {
      this.swap(i, this.getParent(i))
      i = this.getParent(i)
    }
  }
  // 下沉 第i个元素，与左右子节点比较大小
  shiftDown(i) {
    // 下沉到底就停止
    while (this.getLeft(i) < this.size()) {
      // 找出左右节点中较小的那个(大顶堆就是较大的那个)
      let temp = this.getLeft(i)
      let right = this.getRight(i)
      // 右节点存在则比一下大小
      if (this.heap[right] && this.compare(right, temp)) {
        temp = right
      }
      if (this.compare(i, temp)) break // i 已经是最小的了 结束下沉
      this.swap(i, temp)
      i = temp
    }
  }
  // 插入 元素val
  insert(val) {
    this.heap.push(val)
    this.shiftUp(this.heap.length - 1)
  }
  // 移出堆顶，并且重新进行堆化
  poll() {
    this.swap(0, this.heap.length - 1)
    const top = this.heap.pop()
    this.shiftDown(0)
    return top
  }
  // 获取堆顶
  peek() {
    return this.heap[0]
  }
  size() {
    return this.heap.length
  }
}
```

#### 精修版

```js
// 精修版本
class PriorityQueue {
  // 初始化传入比较函数传入
  constructor(data, cmp) {
    this.data = [null].concat(data) // 注意这步操作，是为了方便计算父节点
    this.cmp = cmp // 比较函数
    // 这一步是为什么呢??? todo
    for (let i = data.length >> 1; i > 0; i--) {
      this.down(i)
    }
  }
  get size() {
    return this.data.length - 1
  }
  get top() {
    return this.data[1]
  }
  swap(i, j) {
    ;[this.data[i], this.data[j]] = [this.data[j], this.data[i]]
  }
  // 利用递归进行上浮和下沉
  up(i) {
    if (i === 1) return
    const p = i >> 1
    if (this.cmp(this.data[p], this.data[i])) {
      this.swap(i, p)
      this.up(p)
    }
  }
  down(i) {
    let j = i
    const n = this.data.length
    if (i === n - 1) return
    const l = i << 1,
      r = l + 1
    if (l < n && this.cmp(this.data[i], this.data[l])) i = l
    if (r < n && this.cmp(this.data[i], this.data[r])) i = r
    if (j !== i) {
      this.swap(i, j)
      this.down(i)
    }
  }
  push(val) {
    // 把值加入末尾,然后上浮它
    this.up(this.data.push(val) - 1)
  }
  pop() {
    // 把第一个和最后一个交换, pop出需要的值,把最后一个下沉
    this.swap(1, this.data.length - 1)
    const res = this.data.pop()
    this.down(1)
    return res
  }
}
```
